const fs = require('fs-extra');

// Generate SVG for matrix-like animation
function generateMatrixAnimation() {
  const width = 800;
  const height = 150;
  const numColumns = 40;
  const columnWidth = width / numColumns;
  
  // Generate random digital rain columns
  let columns = '';
  for (let i = 0; i < numColumns; i++) {
    const x = i * columnWidth;
    const y = Math.random() * -height;
    const length = 20 + Math.random() * 30;
    const speed = 0.5 + Math.random() * 1.5;
    const opacity = 0.5 + Math.random() * 0.5;
    
    columns += `
      <g opacity="${opacity}">
        <path class="matrix-column" 
              d="M ${x} ${y} v ${length}" 
              stroke="#66FCF1" 
              stroke-width="1.5"
              stroke-dasharray="${length}"
              stroke-dashoffset="${length * 2}">
          <animate attributeName="stroke-dashoffset" 
                   from="${length * 2}" 
                   to="0" 
                   dur="${speed}s" 
                   repeatCount="indefinite" />
          <animate attributeName="y" 
                   from="${y}" 
                   to="${height}" 
                   dur="${speed * 5}s" 
                   repeatCount="indefinite" />
        </path>
      </g>
    `;
  }
  
  // Generate characters
  let chars = '';
  for (let i = 0; i < 200; i++) {
    const x = Math.random() * width;
    const y = Math.random() * height;
    const char = String.fromCharCode(33 + Math.floor(Math.random() * 94));
    const opacity = 0.3 + Math.random() * 0.7;
    const size = 8 + Math.random() * 6;
    const fadeInDelay = Math.random() * 5;
    
    chars += `
      <text x="${x}" y="${y}" 
            font-family="monospace" 
            font-size="${size}" 
            fill="#66FCF1" 
            opacity="0">
        ${char}
        <animate attributeName="opacity" 
                 from="0" 
                 to="${opacity}" 
                 dur="1s" 
                 begin="${fadeInDelay}s" 
                 fill="freeze" />
        <animate attributeName="opacity" 
                 from="${opacity}" 
                 to="0" 
                 dur="2s" 
                 begin="${fadeInDelay + 2}s" 
                 fill="freeze" />
      </text>
    `;
  }
  
  // Complete SVG
  const svg = `
  <svg width="${width}" height="${height}" xmlns="http://www.w3.org/2000/svg">
    <rect width="100%" height="100%" fill="#0B0C10" />
    ${columns}
    ${chars}
  </svg>
  `;
  
  fs.writeFileSync('animation.svg', svg);
  console.log('Generated matrix animation SVG');
}

generateMatrixAnimation();
